YearMonth = 
Base.new do
  register_xml_types %w[
    gYearMonth
  ]

  def to_ruby(string)
    years, months = string.split '-'
    WsdlMapper::CoreExt::TimeDuration.new years: years.to_i, months: months.to_i
  end

  def to_xml(object)
    '%04d-%02d' % [object.years, object.months]
  end
end
