def link
  schema = WsdlMapper::Dom::Schema.new
  @description.each_schema do |s|
    schema.add_import s.target_namespace, s
  end
  schema_linker = WsdlMapper::DomParsing::Linker.new schema
  schema_linker.link

  link_messages
  link_port_types
  link_bindings
  link_services

  [@description, schema]
end
