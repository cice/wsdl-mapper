def self.get(ns, name)
  @cache ||= Hash.new do |h, k|
    h[k] = Hash.new do |h2, k2|
      h2[k2] = new k, k2
    end
  end

  name = name.to_s unless name.is_a?(String)
  @cache[ns][name]
end
