def initialize(on_nil: nil, &block)
  if block
    @data = Hash.new do |h, k|
      h[k] = Hash.new do |h2, k2|
        h2[k2] = block[WsdlMapper::Dom::Name.get(k, k2)]
      end
    end
  else
    @data = {}
  end
  @on_nil = on_nil
end
