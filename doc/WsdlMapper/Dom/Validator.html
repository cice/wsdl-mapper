def validate_roots
  @schema.each_type do |type|
    next unless type.is_a?(::WsdlMapper::Dom::SimpleType)
    next if type.root.is_a?(::WsdlMapper::Dom::BuiltinType)

    @errors << Error.new(type, :invalid_root)
  end
end
