def stub_action(url, action, body)
  hsh = @actions[url]
  unless hsh
    hsh = @actions[url] = {}
    @stubs.post url do |env|
      req_action = env.request_headers['SOAPAction']
      [200, {}, hsh[req_action]]
    end
  end
  hsh[action] = body
end
